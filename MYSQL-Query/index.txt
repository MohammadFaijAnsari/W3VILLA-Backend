     CREATE DATABASE DATABASE_NAME;
Use Database
     USE DATABASE_NAME
Create table inside the Database
     CREATE TABLE TABLE_NAME(
        f1 datatype(arg) constraint,
        f2 datatype(arg) constraint
    );
Insert Data in the table
   INSERT INTO TABLE_NAME
    (F1,F2,F3...) VALUES('data1','data2',....);
View All Data from Table;
    SELECT * FROM TABLE_NAME;
View All Data for any Specifice Condition
    SELECT * FROM TABLE_NAME WHERE CONDITION;
View Particullar Data from the Table
   SELECT (Specifice Column ) FROM TABLE_NAME WHERE CONDITION;
Update Table Values

Two Table Connect Using MYSQL Query

   CREATE TABLE users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL
   );

   CREATE TABLE orders (
    order_id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    product_name VARCHAR(100) NOT NULL,
    order_date DATE,
    FOREIGN KEY (user_id) REFERENCES users(user_id)   
   );
